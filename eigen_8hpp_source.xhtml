100template <fcarouge::eigen::is_eigen Numerator,
101 fcarouge::eigen::is_eigen Denominator>
102constexpr auto operator/(const Numerator &lhs, const Denominator &rhs)
103 -> fcarouge::eigen::matrix<typename Denominator::Scalar,
104 Numerator::RowsAtCompileTime,
105 Denominator::RowsAtCompileTime> {
106 return rhs.transpose()
107 .fullPivHouseholderQr()
108 .solve(lhs.transpose())
109 .transpose();
110}
